import urllib, urllib.request
url = 'http://export.arxiv.org/api/query?search_query=all:data+compression&start=0&max_results=2'
data = urllib.request.urlopen(url)
text = data.read().decode('utf-8')
print(text)

root = ET.fromstring(text)
print(root)
# Extract Feed Information
feed_info = {
    'link': root.find('link').attrib['href'],
    'title': root.find('title').text,
    'id': root.find('id').text,
    'updated': root.find('updated').text,
    'total_results': root.find('{http://a9.com/-/spec/opensearch/1.1/}totalResults').text,
    'start_index': root.find('{http://a9.com/-/spec/opensearch/1.1/}startIndex').text,
    'items_per_page': root.find('{http://a9.com/-/spec/opensearch/1.1/}itemsPerPage').text
}

# Extract Entries Information
entries = root.findall('entry')
entries_info = []

for entry in entries:
    entry_data = {
        'id': entry.find('id').text,
        'updated': entry.find('updated').text,
        'published': entry.find('published').text,
        'title': entry.find('title').text,
        'summary': entry.find('summary').text.strip(),
        'authors': [author.find('name').text for author in entry.findall('author')],
        'links': [link.attrib['href'] for link in entry.findall('link')],
        'category': [category.attrib['term'] for category in entry.findall('category')]
    }
    entries_info.append(entry_data)

# Format and Print Information

# Format and Print Information
print("Feed Information:")
for key, value in feed_info.items():
    print(f"{key.capitalize()}: {value}")

print("\nEntries Information:")
for idx, entry_info in enumerate(entries_info, start=1):
    print(f"\nEntry {idx}:")
    for key, value in entry_info.items():
        if isinstance(value, list):
            print(f"{key.capitalize()}:")
            for item in value:
                print(f"  - {item}")
        else:
            print(f"{key.capitalize()}: {value}")
